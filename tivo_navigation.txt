# TiVo Navigation Script Configuration
#
# This file defines button sequences for common TiVo navigation tasks.
# Each command is defined with a name and a sequence of button presses.
#
# Format:
#   [command_name]
#   button_name delay_seconds
#   button_name delay_seconds
#   ...
#
# Available buttons:
#   UP, DOWN, LEFT, RIGHT, SELECT, TIVO (home), LIVETV, GUIDE, INFO
#   BACK, PLAY, PAUSE, REVERSE, FORWARD, REPLAY, ADVANCE, RECORD
#   CHANNELUP, CHANNELDOWN, ENTER, CLEAR, NUM0-NUM9
#   TOP (jump to top with 50 rapid UP presses)
#   BOTTOM (jump to bottom with 50 rapid DOWN presses)
#
# Special commands:
#   WAIT_FOR "text" - Wait for text to appear in log file before continuing
#                     Example: WAIT_FOR "Start sending"
#   DELETE_SOURCE_FILE - Delete the transferred file from filesystem
#                        (extracts filename from last "Done sending" log message)
#   LOCATE_SHARE "text" - Navigate through shares until log shows Container matching text
#                         Presses DOWN+SELECT, monitors log, backs out with LEFT if wrong
#                         Example: LOCATE_SHARE "Watcher"
#   TRANSFER_ALL - Loop through all items in current list, pressing SELECT twice on each
#                  Waits for "Start sending" before moving to next item
#
# Delay is in seconds (can be decimal like 0.5)
# Lines starting with # are comments

BOTTOM 0.5

[import-wait-remove]
# Navigate to Import, wait for transfer to complete, then remove file and go back
TIVO 0.5
SELECT 0.5
LEFT 0.5
BOTTOM 0.5
RIGHT 0.5
BOTTOM 0.5
SELECT 0.5
SELECT 0.5
SELECT 0.5
WAIT_FOR "Start sending"
LEFT 1

DELETE_SOURCE_FILE
LEFT 0.5
TIVO 0.5

[watcher]
# Navigate to Import, locate Watcher share, and transfer all items

LEFT 0.5
BOTTOM 0.5
RIGHT 0.5

DELETE_SOURCE_FILE
TIVO 0.5
